276  {
277  if(verbose_)
278  std::cout<<" Solving forward problem "<<std::endl;
279 
281  auto p = Spacy::Newton::Parameter{};
282  p.setRelativeAccuracy(1e-12);
283 
284  // set source term
285  ::Spacy::cast_ref<C1Operator<ForwardFunctionalDefinition> >(A).setSource(u);
286 
288  if(verbose_)
289  std::cout<<"Starting Newton "<<std::endl;
290  auto y = covariantNewton(A,p);
291 
292  //set initial value for next forward solution(A) and for OCP (tsf)
293  // initial data only enters d1, hence not needed to set for normalstepfunctional as d1 is not called
294  auto y_impl = ::Spacy::cast_ref<Vector<DescriptionsY> >(y);
295  if(::Spacy::is<C2Functional<TangentialStepFunctionalDefinition> >(tsf))
296  {
297  auto& tsf_impl = ::Spacy::cast_ref<C2Functional<TangentialStepFunctionalDefinition> >(tsf);
298  auto& A_impl = ::Spacy::cast_ref<C1Operator<ForwardFunctionalDefinition> >(A);
299 
300  tsf_impl.setInitialCondition(boost::fusion::at_c<0>(y_impl.get(gm_forward_.getTempGrid().getDtVec().size()-1).data).coefficients());
301  A_impl.setInitialCondition(boost::fusion::at_c<0>(y_impl.get(gm_forward_.getTempGrid().getDtVec().size()-1).data).coefficients());
302  }
303  return y;
304  }
