150  {
151  using VYSetDescription = Detail::ExtractDescription_t<VariableSetDescription,0>;
152  using VUSetDescription = Detail::ExtractDescription_t<VariableSetDescription,1>;
153  using VPSetDescription = Detail::ExtractDescription_t<VariableSetDescription,2>;
154  using VectorImplY = Vector<VYSetDescription>;
155  using VectorImplU = Vector<VUSetDescription>;
156  using VectorImplP = Vector<VPSetDescription>;
157 
158  using VarSet = typename VariableSetDescription::VariableSet;
159  using PSV = ::Spacy::ProductSpace::Vector;
160  auto x_ps = ::Spacy::cast_ref<PSV>(x);
161 
162  //subvectors as Spacy::Vector
163  auto x_y = (::Spacy::cast_ref<PSV>(x_ps.component(PRIMAL))).component(0);
164  auto x_u = (::Spacy::cast_ref<PSV>(x_ps.component(PRIMAL))).component(1);
165  auto x_p = (::Spacy::cast_ref<PSV>(x_ps.component(DUAL))).component(0);
166 
167  //Implementation on as Spacy::KaskadeParabolic::Vector
168  assert(Spacy::is<VectorImplY>(x_y));
169  auto x_y_impl = ::Spacy::cast_ref<VectorImplY>(x_y);
170  assert(::Spacy::is<VectorImplU>(x_u));
171  auto x_u_impl = ::Spacy::cast_ref<VectorImplU>(x_u);
172  assert(::Spacy::is<VectorImplP>(x_p));
173  auto x_p_impl = ::Spacy::cast_ref<VectorImplP>(x_p);
174 
175  auto vc = ::Spacy::creator<::Spacy::KaskadeParabolic::VectorCreator<VYSetDescription> >(x_y.space());
176  auto gm = vc.getGridMan().getKaskGridMan();
177 
178  auto no_time_steps = gm.size();
179 
180  std::vector<VarSet> v;
181  for(auto i = 0u;i<no_time_steps;i++)
182  {
183  VariableSetDescription varSetDesc(vc.getSpace(i),{"y","u","p"});
184  v.push_back(VarSet(varSetDesc));
185  ::boost::fusion::at_c<0>(v.at(i).data) = boost::fusion::at_c<0>(x_y_impl.get(i).data);
186  ::boost::fusion::at_c<1>(v.at(i).data) = boost::fusion::at_c<0>(x_u_impl.get(i).data);
187  ::boost::fusion::at_c<2>(v.at(i).data) = boost::fusion::at_c<0>(x_p_impl.get(i).data);
188  ::Kaskade::writeVTKFile(gm.at(i)->grid().leafGridView(), v.at(i) ,fileName+std::to_string(i),::Kaskade::IoOptions(),vc.getGridMan().getFEorder());
189  }
190  return;
